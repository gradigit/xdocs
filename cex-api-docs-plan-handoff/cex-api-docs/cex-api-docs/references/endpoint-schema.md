# Endpoint Schema

Every extracted endpoint is stored as a JSON object following this schema. All fields marked
`[SOURCE]` must have a corresponding `source_url` and `crawled_at`. Fields marked `[DERIVED]`
are generated by the parser and clearly labeled as AI-generated.

## Full Schema

```json
{
  "_meta": {
    "schema_version": "1.0",
    "generated_by": "cex-api-docs/cex_parse.py",
    "generated_at": "ISO 8601 timestamp"
  },

  "exchange": "string \u2014 exchange ID from registry",
  "section": "string \u2014 spot|futures_usdm|futures_coinm|portfolio_margin|unified|margin|etc",

  "endpoint": {
    "method": "string \u2014 GET|POST|PUT|DELETE [SOURCE]",
    "path": "string \u2014 /fapi/v1/openOrders [SOURCE]",
    "base_url": "string \u2014 https://fapi.binance.com [SOURCE]",
    "full_url": "string \u2014 base_url + path [DERIVED]",
    "api_version": "string \u2014 v1, v2, v5, etc. [SOURCE]",
    "is_latest_version": "boolean [DERIVED] \u2014 true if this is the latest known version",

    "description": "string \u2014 English description [SOURCE or TRANSLATED]",
    "description_original": "string \u2014 original language if non-English [SOURCE]",

    "parameters": [
      {
        "name": "string [SOURCE]",
        "type": "string \u2014 STRING|INTEGER|DECIMAL|BOOLEAN|ENUM|ARRAY|OBJECT [SOURCE]",
        "required": "boolean [SOURCE]",
        "description": "string [SOURCE]",
        "default": "any | null [SOURCE]",
        "enum_values": ["array of allowed values if type is ENUM [SOURCE]"],
        "example": "any [SOURCE]",
        "notes": "string \u2014 additional constraints, gotchas [SOURCE]"
      }
    ],

    "headers": [
      {
        "name": "string [SOURCE]",
        "required": "boolean [SOURCE]",
        "description": "string [SOURCE]",
        "example": "string [SOURCE]"
      }
    ],

    "required_permissions": {
      "permissions": ["array of permission names [SOURCE]"],
      "permission_notes": "string \u2014 additional context about permissions [SOURCE]",
      "source_url": "string \u2014 exact URL where permissions are documented",
      "confidence": "high|medium|low \u2014 how clearly the docs state this"
    },

    "rate_limit": {
      "weight": "number \u2014 request weight/cost [SOURCE]",
      "weight_conditions": "string \u2014 if weight varies by params [SOURCE]",
      "rate_limit_type": "string \u2014 REQUEST_WEIGHT|ORDER|RAW_REQUESTS|etc [SOURCE]",
      "interval": "string \u2014 1s|1m|5m|1h|etc [SOURCE]",
      "limit": "number \u2014 max requests/weight per interval [SOURCE]",
      "limit_notes": "string \u2014 VIP tiers, special conditions [SOURCE]",
      "ip_or_uid": "string \u2014 ip|uid|both [SOURCE]",
      "source_url": "string"
    },

    "response": {
      "description": "string [SOURCE]",
      "content_type": "string \u2014 application/json [SOURCE]",
      "example": "object \u2014 example response body [SOURCE]",
      "fields": [
        {
          "name": "string [SOURCE]",
          "type": "string [SOURCE]",
          "description": "string [SOURCE]"
        }
      ]
    },

    "error_codes": [
      {
        "code": "string|number [SOURCE]",
        "message": "string [SOURCE]",
        "description": "string [SOURCE]",
        "http_status": "number [SOURCE]",
        "is_endpoint_specific": "boolean \u2014 true if only this endpoint returns it [DERIVED]"
      }
    ],

    "request_example": {
      "curl": "string \u2014 full curl command [SOURCE or DERIVED]",
      "body": "object \u2014 request body example [SOURCE]",
      "query_string": "string [SOURCE]"
    },

    "websocket": {
      "channel": "string \u2014 subscription channel name [SOURCE]",
      "subscribe_message": "object \u2014 subscription payload [SOURCE]",
      "update_message": "object \u2014 example push message [SOURCE]",
      "unsubscribe_message": "object [SOURCE]"
    },

    "canonical": {
      "operation": "string \u2014 canonical operation ID [DERIVED]",
      "confidence": "high|medium|low [DERIVED]",
      "notes": "string \u2014 why this mapping was chosen [DERIVED]"
    },

    "ccxt_reference": {
      "method": "string \u2014 CCXT unified method name",
      "matches_official": "boolean",
      "discrepancies": ["array of noted differences"],
      "ccxt_source_file": "string \u2014 path in CCXT repo",
      "note": "CCXT is a cross-reference only, not source of truth"
    },

    "source": {
      "url": "string \u2014 exact page URL where this endpoint is documented [SOURCE]",
      "crawled_at": "string \u2014 ISO 8601 [SOURCE]",
      "content_hash": "string \u2014 sha256 of source page [SOURCE]",
      "page_title": "string [SOURCE]"
    },

    "notes": "string \u2014 any additional gotchas, undocumented behavior [SOURCE or DERIVED]",
    "tags": ["array of free-form tags for searchability"]
  }
}
```

## Storage Layout

Parsed endpoints are stored at:

```
cex-docs/
\u251c\u2500\u2500 endpoints/
\u2502   \u251c\u2500\u2500 binance/
\u2502   \u2502   \u251c\u2500\u2500 spot/
\u2502   \u2502   \u2502   \u251c\u2500\u2500 GET_api_v3_openOrders.json
\u2502   \u2502   \u2502   \u251c\u2500\u2500 POST_api_v3_order.json
\u2502   \u2502   \u2502   \u2514\u2500\u2500 ...
\u2502   \u2502   \u251c\u2500\u2500 futures_usdm/
\u2502   \u2502   \u2502   \u251c\u2500\u2500 GET_fapi_v1_openOrders.json
\u2502   \u2502   \u2502   \u2514\u2500\u2500 ...
\u2502   \u2502   \u251c\u2500\u2500 _summary.json          # All endpoints for this exchange (index)
\u2502   \u2502   \u251c\u2500\u2500 _rate_limits.json      # Aggregated rate limit info
\u2502   \u2502   \u251c\u2500\u2500 _error_codes.json      # All error codes for this exchange
\u2502   \u2502   \u2514\u2500\u2500 _permissions.json      # All permissions mapped to endpoints
\u2502   \u251c\u2500\u2500 okx/
\u2502   \u2502   \u2514\u2500\u2500 ...
\u2502   \u2514\u2500\u2500 _canonical_map.json        # Cross-exchange canonical operation mapping
\u251c\u2500\u2500 ccxt_ref/
\u2502   \u2514\u2500\u2500 (cloned CCXT repo or extracted reference data)
\u2514\u2500\u2500 ... (doc-crawler storage: raw/, pages/, meta/, db/)
```

## Summary Files

### _summary.json (per exchange)

```json
{
  "exchange": "binance",
  "last_crawled": "2026-02-06T12:00:00Z",
  "sections": {
    "spot": { "endpoint_count": 45, "last_updated": "..." },
    "futures_usdm": { "endpoint_count": 38, "last_updated": "..." }
  },
  "total_endpoints": 120,
  "total_error_codes": 85,
  "api_versions": { "spot": "v3", "futures": "v1" }
}
```

### _canonical_map.json (cross-exchange)

```json
{
  "get_open_orders": {
    "description": "Get all open/active orders",
    "exchanges": {
      "binance": {
        "spot": { "method": "GET", "path": "/api/v3/openOrders" },
        "futures_usdm": { "method": "GET", "path": "/fapi/v1/openOrders" }
      },
      "okx": {
        "unified": { "method": "GET", "path": "/api/v5/trade/orders-pending" }
      },
      "bybit": {
        "unified": { "method": "GET", "path": "/v5/order/realtime" }
      }
    }
  }
}
```
